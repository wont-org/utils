# 快速上手

# npm

全局引入

npm i @wont/utils -S

import utils from '@wont/utils'

utils.getUrlParam()

按需引入（推荐）

import { getUrlParam } from '@wont/utils'

getUrlParam()

# script

如需使用script，需要自行打包上传

<script src="@wont/utils.js" type="text/javascript"></script>

<script>
    window.onload = function () {
        wontUtils.getUrlParam()
    }
</script>
